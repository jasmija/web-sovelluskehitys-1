<script>
  let saldo;
  let korko;
  let korkovuodessa;
  let kasvanutsaldo;

  saldo = prompt('Anna saldo.');
  korko = prompt('Anna vuotuinen korkoprosentti.');

  korkovuodessa = saldo * (korko / 100);
  kasvanutsaldo = korkovuodessa + parseInt(saldo);

  function financial(x) {
    return Number.parseFloat(x).toFixed(2);
  }
  console.log('Vuodessa kertyvä korko ' + financial(korkovuodessa) + '.');
  console.log('Kasvanut saldo ' + financial(kasvanutsaldo) + '.');
</script>
